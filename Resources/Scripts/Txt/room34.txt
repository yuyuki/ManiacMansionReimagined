// room script file

import int GSotherplayerup, GSotherplayerdown;
import function SetPlayer (int charid);

function room_a (){
  // script for room: Player enters screen (before fadein)
  SetMusicVolume (2);
  SetMusicRepeat (1);
  PlayMusic (7);
  SetObjectClickable (0, 0);
  SetObjectClickable (1, 0);
  SetObjectClickable (4, 0);
  init_object (129, 0);
  init_object (130, 1);
  if (GetGlobalInt (242) == 1)
  {
    character [FRED].room = 34;
    character [FRED].x = 340;
    character [FRED].y = 134;
    character [SANDY].room = 34;
    character [SANDY].x = 300;
    character [SANDY].y = 142;
    SetCharacterIgnoreWalkbehinds (SANDY, 1);
    SetCharacterView (SANDY, 76);
    FaceLocation (SANDY, 300, 150);
  }
  else if (GetPlayerCharacter () == POLICE)
  {
    character [GetPlayerCharacter ()].x = 55;
    character [GetPlayerCharacter ()].y = 136;
    ReleaseCharacterView (GetPlayerCharacter ());
    character [FRED].room = 34;
    character [FRED].x = 175;
    character [FRED].y = 131;
    character [SANDY].room = 34;
    character [SANDY].x = 300;
    character [SANDY].y = 142;
    SetCharacterIgnoreWalkbehinds (SANDY, 1);
    SetCharacterView (SANDY, 76);
    FaceDirection (SANDY, DIR_DOWN);
    FaceCharacter (FRED, GetPlayerCharacter ());
    FaceCharacter (GetPlayerCharacter (), FRED);
    ObjectOn (0);
  }
  AlarmSound ();
}

function room_b (){
  // script for room: Player enters screen (after fadein)
  if (GetGlobalInt (242) == 1)
  {
    Wait (25);
    if (character [GetPlayerCharacter ()].inv [4] == 1)
    {
      ObjectOff (0);
      PlaySound (3);
      SetGlobalInt (129, 2); 
    }
    SetGlobalInt (242, 2);
    BeginCutscene (GetPlayerCharacter ());
    MoveCharacterBlocking (FRED, 150, 138, 1);
    SetPlayer (FRED);
    DisplaySpeech (GetPlayerCharacter (), "Purple Tentacle! Invaders!");
    Wait (40);
    DisplaySpeech (GetPlayerCharacter (), "PURPLE TENTACLE!");
    Wait (50);
    DisplaySpeech (GetPlayerCharacter (), "So...");
    DisplaySpeech (GetPlayerCharacter (), "...you've gotten past all my safeguards!");
    DisplaySpeech (GetPlayerCharacter (), "There's only one thing left to do!");
    MoveCharacterBlocking (GetPlayerCharacter (), 438, 132, 0);
    FaceDirection (GetPlayerCharacter (), DIR_UP);
    DisplaySpeech (GetPlayerCharacter (), "I'm arming the self-destruct mechanism.");
    DisplaySpeech (GetPlayerCharacter (), "Only I know the secret code to stop it!");
    PlaySound (20);
    Wait (30);
    PlaySound (23);
    Wait (30);
    PlaySound (21);
    Wait (30);
    PlaySound (23);
    Wait (30);
    PlaySound (20);
    Wait (30);
    PlaySound (21);
    Wait (30);
    PlaySound (22);
    Wait (30);
    DisplaySpeech (GetPlayerCharacter (), "There!");
    SetGlobalInt (312, 1);
    AlarmSound ();
    MoveCharacterBlocking (GetPlayerCharacter (), 158, 138, 0);
    Wait (5);
    DisplaySpeech (GetPlayerCharacter (), "The house will blow up in 2 minutes.");
    DisplaySpeech (GetPlayerCharacter (), "Heh heh heh!");
    MoveCharacterBlocking (GetPlayerCharacter (), 400, 130, 0);
    FaceDirection (GetPlayerCharacter (), DIR_DOWN);
    if (IsKidAlive (GSotherplayerup)) character [GSotherplayerup].room = -1;
    if (IsKidAlive (GSotherplayerdown)) character [GSotherplayerdown].room = -1;
    SetOwnTimer (28, 2350);
    FinishCutscene (0);
  }
  if (GetGlobalInt (242) == 3)
  {
    // FRED is PC
    Wait (45);
    DisplaySpeech (GetPlayerCharacter (), "Oh... what happened?");
    DisplaySpeech (GetPlayerCharacter (), "I feel much better now!");
    DisplaySpeech (GetPlayerCharacter (), "I'm free of this machine's control!");
    DisplaySpeech (GetPlayerCharacter (), "But wait!");
    MoveCharacterBlocking (FRED, 451, 132, 0);
    FaceDirection (GetPlayerCharacter (), DIR_UP);
    Wait (10);
    DisplaySpeech (GetPlayerCharacter (), "The self-destruct sequence was initiated!");
    DisplaySpeech (GetPlayerCharacter (), "I'll try to turn it off.");
    MoveCharacterBlocking (GetPlayerCharacter (), 451, 137, 0);
    FaceDirection (GetPlayerCharacter (), DIR_DOWN);
    Wait (15);
    if (GetGlobalInt (262) == 0)
    {
      DisplaySpeech (GetPlayerCharacter (), "I can't disable it!");
      DisplaySpeech (GetPlayerCharacter (), "The meteor must have control of the entire system!");
      Wait (15);
      FaceDirection (GetPlayerCharacter (), DIR_RIGHT);
      Wait (10);
      DisplaySpeech (GetPlayerCharacter (), "Hey kid!");
      DisplaySpeech (GetPlayerCharacter (), "You've got to help me!");
      DisplaySpeech (GetPlayerCharacter (), "We have to get rid of that meteor once and for all!");
      MoveCharacterBlocking (GetPlayerCharacter (), 400, 130, 0);
      FaceDirection (GetPlayerCharacter (), DIR_DOWN);
      SetGlobalInt (242, 4);
      Wait (25);
      FinishCutscene (0);
    }
    else
    {
      DisplaySpeech (GetPlayerCharacter (), "I did it!");
      SetGlobalInt (327, 1); // Selbstzerstörung abgebrochen
      Wait (15);
      SetCharacterIgnoreWalkbehinds (SANDY, 0);
      FinalScene ();
    }
  }
  else if (GetPlayerCharacter () == POLICE)
  {
    // close door
    Wait (10);
    PlaySound (3);
    ObjectOff (0);
    SetGlobalInt (129, 0);
    Wait (15);
    DisplaySpeech (GetPlayerCharacter (), "Stand aside!");
    MoveCharacterBlocking (GetPlayerCharacter (), 584, 133, 0);
    FaceDirection (GetPlayerCharacter (), DIR_RIGHT);
    Wait (5);
    // open door
    if (GetGlobalInt (130) != 1)
    {
      PlaySound (2);
      ObjectOn (1);
      SetGlobalInt (130, 1);
      Wait (15);
    }
    SetCharacterView (GetPlayerCharacter (), 105);
    Wait (10);
    character [FRED].room = -1;
    SetGlobalInt (242, 0);
    NewRoom (35);
  }
}

function hotspot9_a (){
  // script for hotspot9: Any click on hotspot
  if (any_click_walk_look (614, 141, DIR_DOWN, "'Insert card key to unlock door.'") == 0)
  {
    int result = any_click_use_inv (4, 614, 141, DIR_RIGHT);
    if ((result == 2) && (GetGlobalInt (130) != 1))
    {
      Wait (30);
      PlaySound (23);
      Wait (10);
      MoveCharacterBlocking (GetPlayerCharacter (), 584, 133, 0);
      FaceDirection (GetPlayerCharacter (), DIR_RIGHT);
      Wait (20);
      PlaySound (43);
      ObjectOn (1);
      SetGlobalInt (130, 1);
    }
    else if (result == 0) Unhandled ();
  }    
}
