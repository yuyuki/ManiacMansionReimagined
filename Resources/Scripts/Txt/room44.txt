// room script file

function TranslateGraphic (){
  SetObjectView (0, TranslateNumber ("_tv_44_view_ 81"));
}

function on_call (int value){
  if (value == GAME_RESTORED)
  {
    TranslateGraphic ();
  }
}

function room_b (){
  // script for room: Player enters screen (before fadein)
  SetCharacterView (GetPlayerCharacter (), 105);
  character [MARK].room = 44;
  character [MARK].x = 148;
  character [MARK].y = 122;
  FaceDirection (MARK, DIR_DOWN);
  TranslateGraphic ();
  AnimateObject (0, 0, 0, 1);
  AlarmSound ();
}

function room_c (){
  // script for room: Player enters screen (after fadein)
  BeginCutscene (NO_CHAR);
  Wait (20);
  DisplaySpeech (MARK, "Hi all you budding writers, musicians and game designers!");
  DisplaySpeech (MARK, "I'm Mark Eteer and my company will publish ANYTHING!");
  DisplaySpeech (MARK, "Just mail it to the address shown below.");
  Wait (30);
  DisplaySpeech (MARK, "Don't forget the stamp!");
  Wait (20);
  FinishCutscene (0);
  character [MARK].room = 37;
  character [MARK].x = 280;
  character [MARK].y = 141;
  ReleaseCharacterView (GetPlayerCharacter ());
  FaceDirection (GetPlayerCharacter (), DIR_UP);
  NewRoom (8);
}
