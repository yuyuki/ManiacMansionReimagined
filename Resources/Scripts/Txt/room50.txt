// room script file
#define b_mod  971
#define b_sel  974
#define bern   4
int timer = 0;
int face = -1;
import int GSotherplayerup, GSotherplayerdown;
int players = 0;

function OtherPlayer (int charid, int objnumber, string description){
  PlaySound (8);
  if (IsObjectOn (objnumber) == 0)
  {
    if (GSotherplayerup < 0)          GSotherplayerup = charid;
    else if (GSotherplayerdown < 0)   GSotherplayerdown = charid;
    else                              charid = -1;
    SetLabelText (12, 0, description);
    if (charid >= 0)
    {
      ObjectOn (objnumber);
      if (charid == bern)
      {
        SetObjectGraphic (7 + bern, b_sel);
        ObjectOn (7 + bern);
        if (face == bern) face = -1;
      }
      players++;
    }
  }
  else
  {
    ObjectOff (objnumber);
    SetLabelText (12, 0, "");
    if (GSotherplayerup == charid)        GSotherplayerup = -1;
    else if (GSotherplayerdown == charid) GSotherplayerdown = -1;
    else                                  charid = -1;
    if (charid == bern)
    {
      ObjectOff (7 + bern);
      SetObjectGraphic (7 + bern, b_mod);
    }
    players--;
  }
}

#sectionstart room_b  // DO NOT EDIT OR REMOVE THIS LINE
function room_b (){
  // script for room: Repeatedly execute
  if (players == 0) SetLabelText (11, 0, "Please select two other kids.");
  else if (players == 1) SetLabelText (11, 0, "Please select one other kid.");
  else SetLabelText (11, 0, "");
  // change faces
  if (timer > 0) timer--;
  else
  {
    if (face < 0)
    {
      // verändern
      face = Random (6);
      if ((face == 4) && (IsObjectOn (7 + face))) face = -1;
      else ObjectOn (7 + face);
      timer = 30;
    }
    else
    {
      // wiederherstellen
      ObjectOff (7 + face);
      face = -1;
      timer = 80;
    }
  }
}
