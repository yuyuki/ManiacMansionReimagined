// text script for room

import function KeyPadCheckNumber (int number);

// sprites of the digits must be ordered 0 to 9
// digit_graphics is the slot number of digit 0
#define digit_graphics 442

int digits = 0, number = 0;

function press_button (int digit){
  PlaySound (8);
  SetCursorMode (1);
  int do = digits;
  // show another digit
  ObjectOn (do);
  // shift digits to left
  while (do > 0)
  {
    SetObjectGraphic (do, GetObjectGraphic (do - 1));
    do--;
  }
  // set its graphic according the button's number
  SetObjectGraphic (do, digit_graphics + digit);
  digits++;
  number = number * 10 + digit;
  if (digits >= 4)
  {
    // wait so that the player can see the last digit
    Wait (40);
    ObjectOff (0);
    ObjectOff (1);
    ObjectOff (2);
    ObjectOff (3);
    Wait (40);
    // check if correct number
    KeyPadCheckNumber (number);  
  }
}

#sectionstart room_b  // DO NOT EDIT OR REMOVE THIS LINE
function room_b (){
  // script for room: Player enters screen (before fadein)
  GUIOff (0);
  GUIOff (1);
  SetCursorMode (1);
  digits = 0;
  ObjectOff (0);
  ObjectOff (1);
  ObjectOff (2);
  ObjectOff (3);
  number = 0;
  AlarmSound ();
}
#sectionend room_b  // DO NOT EDIT OR REMOVE THIS LINE
